指令 ​

1. 自定义局部指令 ​

  1. 指令其实就是一个对象，对象里面有一堆方法，那一堆方法就是生命周期钩子，在特定情况下会被自动调用。

  2. 每一个生命周期都有四个参数，第一个是 el：被绑定的元素，binding：是一个对象包含了指令的参数，值，历史的值，指令的实例等。后面两个是虚拟 dom，vnode：绑定元素的底层虚拟 dom，prevVnode：前一版的虚拟 dom。

  3. v-MyDirective:100.str="'hello'" ，100 对应的是 binding 对象的 arg，.str 对应的是 binding 对象的 modifiers(修饰符) 对象里面的值，hello 对应的 binding 对象的 value。

vue
<template>
  <div>
    <div v-MyDirective:100.str="'hello'">测试测试测试测试测试测试测试测试</div>
  </div>
</template>

<script setup lang="ts">
// 自定义局部指令
import { Directive } from "vue";

// 命名方式优先使用v开头，后面是指令的名称
const vMyDirective: Directive = {
  // 在绑定元素的 attribute 前 或事件监听器应用前调用
  created(el, binding, vnode, prevVnode) {},
  // 在元素被插入到 DOM 前调用
  beforeMount(el, binding, vnode, prevVnode) {},
  // 绑定元素的父组件以及他自己所有的子节点都挂在完成后调用
  mounted(el, binding, vnode, prevVnode) {
    console.log(el, binding, vnode, prevVnode);
    el.style.color = "red";
  },
  // 绑定元素的父组件更新前调用
  beforeUpdate(el, binding, vnode, prevVnode) {},
  // 绑定元素的父组件及他自己的所有子节点都更新后调用
  updated(el, binding, vnode, prevVnode) {},
  // 绑定元素的父组件卸载前调用
  beforeUnmount(el, binding, vnode, prevVnode) {},
  // 绑定元素的父组件卸载后调用
  unmounted(el, binding, vnode, prevVnode) {},
};
</script>

3. 多个指令的安装 ​

  1. src 文件下新建文件夹 directives ，directives文件夹中新建index.ts文件, 新建modules文件夹 如图：

  2. modules 里面每一个文件就是一个指令。示例代码如下:

ts
import { Directive } from "vue";

export default {
  mounted(el) {
    el.style.backgroundColor = "lightblue";
  },
} as Directive;
  1. index.ts 文件里面的内容如下:
ts
import { App, Directive } from "vue";

type DirectivesObjType = {
  default: Directive;
};

const directives = import.meta.glob("./modules/*.ts", {
  eager: true /* 同步导入 */,
}) as Record<string, DirectivesObjType>;

export default {
  install(app: App) {
    Object.keys(directives).forEach((key) => {
      // 获取文件名
      const lastIndex = key.lastIndexOf("/");
      const directiveName = key.slice(lastIndex + 1, key.length - 3);

      // 注册全局指令
      app.directive(directiveName, directives[key].default);
    });
  },
};
  1. 在 main.ts 中 use 指令
ts
import { createApp } from "vue";
import App from "./App.vue";
import directive from "./directives/index";
const app = createApp(App);

// 安装全局指令
app.use(directive);

app.mount("#app");
  1. 使用指令
vue
<template>
  <div>
    <div v-border-directive>测试</div>

    <input type="text" v-focus-directive />

    <div v-style-directive>测试2</div>

    <div v-style-bg-directive>测试3</div>
  </div>
</template>

2. 子传父 ​

父组件

vue
<template>
  <div style="border: 5px solid lightseagreen; padding: 10px">
    <h1>父组件</h1>
    <AboutView :data="data" @changeData="changeData" />
  </div>
</template>

<script setup lang="ts">
import AboutView from "./views/AboutView.vue";
import { ref } from "vue";
const data = ref(100);

// 子组件触发的函数
const changeData = (d: number) => {
  data.value = d;
};
</script>

子组件

vue
<template>
  <div style="border: 1px solid lightblue; margin: 10px">
    <h1>子组件2-AboutView</h1>
    <h2>{{ props.data }}</h2>

    <button @click="onChangeData">change data</button>
  </div>
</template>

<script setup lang="ts">
// 给 props 一个默认值的方式
const props = withDefaults(
  defineProps<{
    data: number;
  }>(),
  {
    data: 1,
  }
);

// 子传父
// ts 专属写法
interface EmitsType {
  (e: "changeData", data: number): void;
}
const emits = defineEmits<EmitsType>();

//非ts写法
// const emits = defineEmits(['changeData'])

const onChangeData = () => {
  const newData = props.data + 1;
  emits("changeData", newData);
};
</script>

3. expose & ref ​

父组件

vue
<template>

    <HomeView :data="data" ref="homeViewRef" />
    <button @click="onGetSonData">获取子组件数据</button>
  </div>
</template>

<script setup lang="ts">
import HomeView from "./views/HomeView.vue";
import { ref } from "vue";
const data = ref(100);

// 2. 定义ref
const homeViewRef = ref<InstanceType<typeof HomeView>>();
const onGetSonData = () => {
  // 3. 调用暴露出来的方法，拿到数据
  console.log(homeViewRef.value?.getSonData());
};
</script>

5. v-model ​

父组件

vue
<template>
  <div style="border: 5px solid lightseagreen; padding: 10px">
    <h1>父组件</h1>
    <h2>data:{{ data }}</h2>
    <h2>info:{{ info }}</h2>
    <!-- 多个 v-model的使用 -->
    <HomeView v-model="data" v-model:info="info" />
  </div>
</template>

<script setup lang="ts">
import HomeView from "./views/HomeView.vue";
import { ref } from "vue";
const data = ref(100);
const info = ref<string>("info");
</script>

子组件

vue
<template>
  <div style="border: 2px solid lightgreen; margin: 10px">
    <h1>子组件1- HomeView</h1>
    <hr />
    <div>{{ props.modelValue }}</div>
    <input type="number" @input="onInput" />
    <input type="text" @change="onChange" />
  </div>
</template>

<script setup lang="ts">
const props = defineProps({
  modelValue: {
    type: Number,
    default: 123,
  },
});

// ts 写法
interface EmitsType {
  (e: "update:modelValue", value: number): void;
  (e: "update:info", value: string): void;
}
const emits = defineEmits<EmitsType>();

// 非ts 写法
// const emits = defineEmits(['update:modelValue'])

const onInput = (e: Event) => {
  let target = e.target as HTMLInputElement;
  // v-model
  emits("update:modelValue", Number(target.value));
};
const onChange = (e: Event) => {
  let target = e.target as HTMLInputElement;
  // v-model：info
  emits("update:info", target.value);
};
</script>

6. provide & inject ​

ts
// 注意这里需要单独放在其他文件中，因为setup中不能export
export const symbolKey = Symbol();
  1. 父组件
vue
<template>
  <div style="border: 5px solid lightseagreen; padding: 10px">
    <h1>父组件</h1>
    <h2>age:{{ age }}</h2>
    <HomeView />
    <button @click="onChangeAge">change age</button>
  </div>
</template>

<script setup lang="ts">
import HomeView from "./views/HomeView.vue";
import { symbolKey } from "./tools/symbol";
import { provide, readonly, ref } from "vue";

provide("name", "zrs");

// 如果注入的是响应式数据，则会保持响应式
const age = ref(18);
provide("age", age);

const onChangeAge = () => {
  age.value++;
};

const readonlyValue = ref(1999999);
// 想要提供的数据不被更改
provide("readonlyValue", readonly(readonlyValue));

provide(symbolKey, "symbolValue111111");
</script>
  1. 后代组件
vue
<template>
  <div style="border: 1px solid lightgreen; padding: 20px">
    <h2>子组件</h2>
    <h2>name:{{ name }}</h2>
    <h2>age:{{ age }}</h2>
    <h2>sex:{{ sex }}</h2>
    <h2>readonlyValue:{{ readonlyValue }}</h2>
    <h2>symbolValue:{{ symbolValue }}</h2>

    <button @click="onChangeReadonlyValue">change readonlyValue</button>
    <button @click="onChangeAge">change age</button>
  </div>
</template>

<script setup lang="ts">
import { inject, type Ref } from "vue";
import { symbolKey } from "@/tools/symbol";

const name = inject<Ref<string>>("name");

const age = inject<Ref<number>>("age");

// 为了避免在用不到默认值的情况下进行不必要的计算或产生副作用
// 我们可以使用工厂函数来创建默认值
// 第二个参数是默认值,并且是使用工厂函数创建的
// 第三个参数表示默认值应该被当作一个工厂函数
const sex = inject("sex", () => "man", true);

// 指定接收注入的值的类型
const readonlyValue = inject<Ref<number>>("readonlyValue");

// 只读的注入不允许被修改
const onChangeReadonlyValue = () => {
  // Set operation on key "value" failed: target is readonly.
  if (readonlyValue) readonlyValue.value = 39999;
};

const onChangeAge = () => {
  // 修改了之后，会影响父组件里的值
  if (age) age.value = 200;
};

const symbolValue = inject(symbolKey);
</script>

7. bus 通信 ​

  1. 安装插件
pnpm
pnpm i mitt -S
  1. 使用插件
ts
import mitt from "mitt";

type Events = {
  foo: string;
  bar: number;
};
export const bus = mitt<Events>();
  1. 组件 1
vue
<template>
  <div style="border: 1px solid lightblue; padding: 10px">
    <h1>父组件</h1>
    <div>info:{{ info }}</div>
    <div>num:{{ num }}</div>
    <HomeView />
    <button @click="onClick">关闭 监听</button>

    <button @click="onCloseAll">清除 所有 监听</button>
  </div>
</template>

<script setup lang="ts">
import { ref } from "vue";
import HomeView from "./views/HomeView.vue";
import { bus } from "@/tools/bus";

const info = ref<string>("");
const num = ref<number>(0);

bus.on("foo", (data: string) => {
  console.log("data", data);
  info.value = data;
});

bus.on("bar", (data: number) => {
  console.log("data", data);
  num.value = data;
});

// 关闭一个监听
const onClick = () => {
  bus.off("foo");
};

// 清除所有监听
const onCloseAll = () => {
  bus.all.clear();
};
</script>
  1. 组件 2
vue
<template>
  <div style="border: 2px solid lightgreen; padding: 10px">
    <h1>子组件-homeView</h1>
    <button @click="onClick">send info</button>
    <button @click="onClickBar">send num</button>
  </div>
</template>

<script setup lang="ts">
import { bus } from "@/tools/bus";

const onClick = () => {
  bus.emit("foo", "homeView");
};

const onClickBar = () => {
  bus.emit("bar", 123);
};
</script>
