    <script>
      // 创建对象
      const obj = new Object();
      console.log(obj);

      // 创建对象2
      const obj2 = {};

      // Object 上的静态方法

      // 1. 合并对象
      const obj3 = { name: "张三" };
      const obj4 = { age: 18 };
      const obj5 = { sex: "男" };
      const newObj = {}; // 目标对象
      Object.assign(newObj, obj3, obj4, obj5); // 将后面的对象合并到第一个对象中，浅拷贝
      console.log(newObj);

      // 2. 以某个对象为原型，创建一个新的对象
      const obj6 = {
        key: 1,
        getKey: function () {
          return this.key;
        },
      };

      // 创建一个新对象，新对象的原型是obj6，
      // 第二个参数是向新对象中添加属性，并且添加属性描述
      const obj7 = Object.create(obj6, {
        key2: {
          value: "2",
          writable: false,
          configurable: false,
          enumerable: false,
        },
      });
      delete obj7.key2;
      console.log(obj7);
      console.log(obj7.getKey());

      // 3.  向一个对象中添加一个属性
      const obj8 = {};
      Object.defineProperty(obj8, "name", {
        enumerable: false, // 属性描述符 不可枚举
        value: "张三", // 属性的值
        writable: false, // 不可修改
        configurable: false, // 不可删除
      });
      console.log(obj8);
      delete obj8.name;
      console.log(obj8);

      // 4. 向一个对象中添加多个属性
      Object.defineProperties(obj8, {
        age: {
          enumerable: true, // 只有这个值为true ，才能for in 遍历出来
          value: 18,
          writable: true,
          configurable: true,
        },
        sex: {
          value: "男",
          writable: true,
          configurable: true,
          enumerable: true,
        },
      });
      console.log(obj8);

      // 5. 返回一个二维数组，将对象的属性名和属性值作为数组的元素
      console.log(Object.entries(obj8));
      const newArr = Object.entries(obj8);

      // 5.1 将二维数组转换为对象
      const o = Object.fromEntries(newArr);
      console.log("o", o);

      // 6. 冻结一个对象，其他代码不能删除或更改其任何属性。是js对象最高级的保护措施。
      const obj9 = {
        name: "张三",
        age: 18,
        sex: "男",
      };
      Object.freeze(obj9);
      console.log(obj9.name);
      obj9.name = "李四"; // 改行代码没有效果，只有严格模式才会报错
      console.log(obj9.name);

      // 6.1 判断一个对象是否被冻结, true 被冻结  false 没有被冻结
      console.log("冻结？", Object.isFrozen(obj9));
      console.log("冻结？", Object.isFrozen(obj8));

      // 7. 得到一个属性名的描述符状态
      // 因为obj9 被冻结了，所以他的writable和configurable全是false
      console.log(Object.getOwnPropertyDescriptor(obj9, "name"));
      // 7.1 返回一个对象所有自身属性的描述符
      console.log(Object.getOwnPropertyDescriptors(obj9));

      // 8. 得到对象自身的所有属性名数组，不管该属性能不能被枚举，但是symbol的属性名获取不到
      const age = Symbol("age");
      const obj10 = {
        name: "zzr",
        [age]: 18,
      };
      Object.defineProperty(obj10, "sex", {
        value: "nan",
        enumerable: false,
        configurable: false,
        writable: false,
      });
      console.log(Object.getOwnPropertyNames(obj10));

      // 9. 获取对象上所有的symbol类型的属性名的数组
      console.log(Object.getOwnPropertySymbols(obj10));

      // 10. 获取指定对象的原型
      console.log(Object.getPrototypeOf(obj7));

      // 传入一个属性名，如果该属性名是自身的，则返回true，是继承或者不存在则返回false
      // 这个是静态方法
      const obj11 = {
        name: "zrs",
      };
      console.log(Object.hasOwn(obj11, "name")); // true
      console.log(Object.hasOwn(obj11, "age")); // false

      // 这个是 -实例- 上的方法
      console.log(obj11.hasOwnProperty("name")); // true
      console.log(obj11.hasOwnProperty("age")); // false

      // 11. 密封一个对象：不能添加新属性、不能删除现有属性或更改其可枚举性和可配置性、不能重新分配其原型。
      // 只要现有属性的值是可写的，它们仍然可以更改。

      const obj12 = {
        name: "zrs",
      };
      Object.seal(obj12);
      obj12.age = 18;
      console.log(obj12);
      obj12.name = "zzr"; // 可写
      console.log(obj12);

      // 11.1 判断一个对象是否可被密封,true:表示对象已经密封，不可扩展，可写的只可以写
      console.log(Object.isSealed(obj12));

      // 12. 防止新属性被添加到对象中
      const obj13 = {
        name: "zrs1111",
        age: 18,
      };

      Object.preventExtensions(obj13);
      obj13.gender = "male";
      obj13.name = "zzaaaaaaaaaaaaar"; // 还是可读写的可读写
      console.log(obj13.name);
      console.log("obj13", obj13);

      // 12.1 判断一个对象是否可扩展，true 表示可扩展，false表示不可扩展
      console.log(Object.isExtensible(obj13));

      // 13. 得到给定对象自有可枚举字符串属性名称的数组。
      console.log(Object.keys(obj13));

      // 14. 返回一个给定对象的自有可枚举字符串键属性值组成的数组。
      console.log(Object.values(obj13));
    </script>
