js 判断数据类型的方式有那些？ ​

1.typeof ​

typeof 只适用于基本数据类型的判定

js
let num = 100;
let str = "hello";
let bool = true;
let arr = [1, 2, 3];
let obj = { name: "zrs", age: 18 };
console.log(typeof num); // number
console.log(typeof str); // string
console.log(typeof bool); // boolean
console.log(typeof arr); // object
console.log(typeof obj); // object
console.log(typeof undefined); // undefined
console.log(typeof null); // object

2. instanceof ​

instanceof 只可以用来判断引用数据类型

js
let arr1 = [1, 2, 3];
let people = {
  name: "zrs",
  age: 18,
};
function func() {
  return "hello";
}
console.log(arr1 instanceof Array); // true
console.log(people instanceof Object); // true
console.log(func instanceof Function); // true

console.log(num instanceof Number); // false
console.log(str instanceof String); // false
console.log(bool instanceof Boolean); // false

3. Object.prototype.toString.call() ​

Object.prototype.toString.call() 判断数据类型，有点麻烦

js
let num1 = 100;
let str1 = "hello";
let bool1 = true;
let arr2 = [1, 2, 3];
let obj1 = { name: "zrs", age: 18 };
function func2() {
  return "hello";
}
console.log(Object.prototype.toString.call(num1)); // [object Number]
console.log(Object.prototype.toString.call(str1)); // [object String];
console.log(Object.prototype.toString.call(bool1)); // [object Boolean];
console.log(Object.prototype.toString.call(arr2)); // [object Array];
console.log(Object.prototype.toString.call(obj1)); // [object Object];
console.log(Object.prototype.toString.call(func2)); // [object Function];

只是知道，没实际运用过

class A {}

js 修改 this 指向的函数 ​

js
const Person = {
  name: "zrs",
  age: 18,
};

function getName(a, b) {
  console.log("a+b", a + b);
  return this.name;
}

// 第一个参数是需要修改this指向的对象，后面就是传参的方式，call 可以挨个传参
console.log("first", getName.call(Person, 1, 2));

// apply 需要将参数作为数组传参
console.log("first", getName.apply(Person, [1, 3]));

// 将需要指向的对象传入，会返回新的函数，依次传参即可
console.log("first", getName.bind(Person)(2, 3));

原型与原型链 ​

js
function Person(name, age) {
  this.name = name;
  this.age = age;
}

const p = new Person("张三", 18);

console.log(p.__proto__);
console.log("Person", Person.prototype);
console.log("per", Person.prototype.__proto__);

原型与原型链：

  1. 创建一个构造函数 构造函数会有一个 prototype 属性，指向原型对象，原型对象会有一个 constructor 属性指向构造函数。
  2. 通过构造函数实例化的时候，实例会有一个 proto 属性，指向构造函数的原型对象。
  3. 当我们调用实例对象的方法和属性的时候，先在自身找，然后去通过原型链一级一级往上找
  4. 原型链的尽头是 null

暂时性死区 ​

js
let num = 1000;
{
  num = 12; // Uncaught ReferenceError: Cannot access 'num' before initialization
  let num;
  console.log("num", num);
}
console.log("num", num);
  1. let 和 const 声明的变量不会被提升，var 声明的会提升。
  2. 上面的代码中，先在全局作用域中声明了变量 num，又在块级作用域中声明了变量 num，后者就被绑定在了这个块级作用域，然后报错的哪一行代码在声明之前使用了局部作用域的变量，所以就报错了。

普通函数与箭头函数 ​

  1. 语法简洁
js
const func1 = () => ({}) // 返回一个对象
const func2 = name => name + '666' // 只有一个参数的情况
const func3 = (a, b) => a + b // 多个参数的情况
  1. 箭头函数没有自己的this，在箭头函数中使用this，其实是与其外部上下文的this保持一致的 ,箭头函数的this指向是固定的，this一旦被绑定，就不会再被任何方式所改变
js
const person = {
    name: 'zrs',
    age: 18,
    say: () => {
        console.log(this.name, this.age)
        console.log('this', this) // this 指向window
    },
    say2: function () {
        console.log(this.name, this.age)
        console.log('this', this) // this 指向 person 实例
    }
}
person.say() // undefined
person.say2() // zrs 18
  1. 箭头函数没有自己arguments对象
js
const func4 = (a, b, c) => {
    console.log(arguments) // arguments is not defined
}
// func4(1, 2, 3)

function func5(a, b, c) {
    console.log('arguments', arguments) // Arguments(3)
}
func5(1, 2, 3)

防抖与节流 ​

html
<button id="btn1">防抖</button> <button id="btn2">节流</button>

1.防抖 ​

js
const btn1 = document.getElementById("btn1");

btn1.addEventListener(
  "click",
  debounce((e) => {
    console.log("e", e);
  }, 1000)
);

// 防抖:频繁触发，但是最后一次生效
function debounce(fn, delay) {
  let timer = null;

  return function () {
    if (timer) {
      clearTimeout(timer);
    }

    timer = setTimeout(() => {
      fn.apply(this, arguments);
    }, delay);
  };
}

3. 节流 ​

js
const btn2 = document.getElementById("btn2");

btn2.addEventListener(
  "click",
  throttle((e) => {
    console.log("e", e);
  }, 1000)
);

// 节流:第一次触发生效，频繁触发，需要间隔时间到了，然后第一次又生效
function throttle(fn, delay) {
  let nowTime = Date.now();

  return function () {
    let curTime = Date.now();

    if (curTime - nowTime > delay) {
      fn.apply(this, arguments);
      nowTime = Date.now();
    }
  };
}
